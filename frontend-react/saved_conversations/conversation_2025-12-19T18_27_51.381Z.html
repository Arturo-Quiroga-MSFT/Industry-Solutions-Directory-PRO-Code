SELECT DISTINCT solutionName, orgName, industryName, solutionAreaName, solutionDescription
FROM dbo.vw_ISDSolution_All
WHERE solutionStatus = 'Approved'
  AND industryName = 'Financial Services'
  AND (
        solutionName LIKE '%core bank%'
        OR solutionDescription LIKE '%core bank%'
        OR solutionName LIKE '%bank modern%'
        OR solutionDescription LIKE '%bank modern%'
      )
ORDER BY solutionName;
